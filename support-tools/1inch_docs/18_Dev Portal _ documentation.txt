const axios = require("axios");

async function httpCall() {
  const url = "https://api.1inch.dev/fusion-plus/relayer/v1.0/submit/secret";

  const config = {
    headers: undefined,
    params: {},
    paramsSerializer: {
      indexes: null,
    },
  };
  const body = {
    secret: "string",
    orderHash: "string",
  };

  try {
    const response = await axios.post(url, body, config);
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}
