const axios = require("axios");

async function httpCall() {
  const url = "https://api.1inch.dev/price/v1.1/1/{addresses}";

  const config = {
    headers: undefined,
    params: {},
    paramsSerializer: {
      indexes: null,
    },
  };

  try {
    const response = await axios.get(url, config);
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}
