const axios = require("axios");

async function httpCall() {
  const url = "https://api.1inch.dev/token/v1.3/search";

  const config = {
    headers: undefined,
    params: {
      query: "1inch",
      limit: 10,
    },
    paramsSerializer: {
      indexes: null,
    },
  };

  try {
    const response = await axios.get(url, config);
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}
