const axios = require("axios");

async function httpCall() {
  const url = "https://api.1inch.dev/nft/v2/contract";

  const config = {
    headers: undefined,
    params: {
      chainId: "1",
    },
    paramsSerializer: {
      indexes: null,
    },
  };

  try {
    const response = await axios.get(url, config);
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}
