const axios = require("axios");

async function httpCall() {
  const url = "https://api.1inch.dev/portfolio/portfolio/v5.0/general/chart";

  const config = {
    headers: undefined,
    params: {
      chain_id: "1",
    },
    paramsSerializer: {
      indexes: null,
    },
  };

  try {
    const response = await axios.get(url, config);
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}
