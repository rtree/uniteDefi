const axios = require("axios");

async function httpCall() {
  const url = "https://api.1inch.dev/token/v1.4/1";

  const config = {
    headers: undefined,
    params: {
      provider: "1inch",
      country: "US",
    },
    paramsSerializer: {
      indexes: null,
    },
  };

  try {
    const response = await axios.get(url, config);
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}
