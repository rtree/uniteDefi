  const config = {
    headers: undefined,
    params: {
      chain_id: "1",
      timerange: "1year",

    paramsSerializer: {
      indexes: null,
    },
  };

  try {
    const response = await axios.get(url, config);
    console.log(response.data);
  } catch (error) {
    console.error(error);

}
