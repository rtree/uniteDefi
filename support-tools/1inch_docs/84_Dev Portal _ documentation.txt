const axios = require("axios");

async function httpCall() {
  const url =
    "https://api.1inch.dev/history/v2.0/history/0x111111111117dC0aa78b770fA6A738034120C302/events/swaps";

  const config = {
    headers: undefined,
    params: {},
    paramsSerializer: {
      indexes: null,
    },
  };
  const body = {
    filter: {
      chain_ids: ["string"],
      token_address_from: {},
      token_address_to: {},
      limit: 0,
    },
  };

  try {
    const response = await axios.post(url, body, config);
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}
