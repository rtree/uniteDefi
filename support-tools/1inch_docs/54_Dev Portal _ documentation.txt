const axios = require("axios");

async function httpCall() {
  const url = "https://api.1inch.dev/swap/v6.1/1/approve/transaction";

  const config = {
    headers: undefined,
    params: {
      tokenAddress: "0x111111111117dc0aa78b770fa6a738034120c302",
      amount: "100000000000",
    },
    paramsSerializer: {
      indexes: null,
    },
  };

  try {
    const response = await axios.get(url, config);
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}
